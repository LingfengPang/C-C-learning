## 栈的介绍

### 20. 有效的括号
```cpp
//O(n) O(n)
class Solution {
public:
    bool isValid(string s) {
        stack<char> st;
        char c;
        for(int i = 0;i<s.size();i++){
            c = s[i];
            if(c == '(' || c == '{' || c == '[')
                st.push(c);
            else if(c == ')' &&st.size()>0&& st.top() == '('){
                    st.pop();
            }
            else if(c == ']' &&st.size()>0&& st.top() == '['){
                    st.pop();
            }
            else if(c == '}' &&st.size()>0&& st.top() == '{'){
                    st.pop();
            }
            else 
                return false;
        }
        return st.empty();
    }
};
```
